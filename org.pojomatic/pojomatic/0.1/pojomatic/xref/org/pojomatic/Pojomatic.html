31    private static SelfPopulatingMap<Class<?>, Pojomator<?>> POJOMATORS =
32      new SelfPopulatingMap<Class<?>, Pojomator<?>>() {
33        @Override
34        @SuppressWarnings("unchecked")
35        // compiler does not know that the type parameter to Pojomator is the same as the type
36        // parameter to Class
37        protected Pojomator<?> create(Class<?> key) {
38          return new PojomatorImpl(key);
39        }
40    };

44    /**
45     * Compute the {@code toString} representation for a pojo.
46     * @param <T> the type of the pojo
47     * @param pojo the pojo - must not be null
48     * @return the {@code toString} representation of {@code pojo}.
49     * @see Pojomator#doToString(Object)
50     */
51    public static <T> String toString(T pojo) {
52      return pojomator(getClass(pojo)).doToString(pojo);
53    }

55    /**
56     * Compute the {@code hashCode} for a pojo.
57     * @param <T> the type of the pojo
58     * @param pojo the pojo - must not be null
59     * @return the {@code hashCode} for {@code pojo}.
60     * @see Pojomator#doHashCode(Object)
61     */
62    public static <T> int hashCode(T pojo) {
63      return pojomator(getClass(pojo)).doHashCode(pojo);
64    }

66    /**
67     * Compute whether {@code pojo} and {@code other} are equal to each other in the sense of
68     * {@code Object}'s {@code equals} method.
69     * @param <T> the type of the pojo
70     * @param pojo the pojo - must not be null
71     * @param other the object to compare to for equality
72     * @return whether {@code pojo} and {@code other} are equal to each other in the sense of
73     * {@code Object}'s {@code equals} method.
74     * @see Pojomator#doEquals(Object, Object)
75     */
76    public static <T> boolean equals(T pojo, Object other) {
77      return pojomator(getClass(pojo)).doEquals(pojo, other);
78    }

80    /**
81     * Get the {@code Pojomator} for {@code pojoClass}.  The same instance will be returned every time
82     * for a given value of {@code pojoClass}.
83     * @param <T> the type represented by {@code pojoClass}
84     * @param pojoClass the class to create a {@code Pojomator} for.
85     * @return a {@code Pojomator<T>}
86     */
87    @SuppressWarnings("unchecked") // compiler does not know that the type parameter to Pojomator is T
88    public static <T> Pojomator<T> pojomator(Class<T> pojoClass) {
89      return (Pojomator<T>) POJOMATORS.get(pojoClass);
90    }
91  
92    @SuppressWarnings("unchecked") // Since Object.getClass returns Class<?>
93    private static <T> Class<T> getClass(T pojo) {
94      return (Class<T>) pojo.getClass();
